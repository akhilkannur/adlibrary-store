Setting Up Your Competitive Tracking System

Competitive Intelligence Framework

Our framework organizes competitive data into actionable insights:

1. Spend Estimation Model

While LinkedIn doesn't show spend, we estimate based on ad volume and duration:

def estimate_spend_level(ad_count, ad_duration_days):
    """
    Estimate competitor ad spend based on observable metrics.
    
    Args:
        ad_count: Number of active ads
        ad_duration_days: Average days ads have been running
    
    Returns:
        Spend level (Low, Medium, High, Very High)
    """
    # Base calculation on ad volume
    if ad_count >= 50:
        volume_factor = 4  # Very High
    elif ad_count >= 20:
        volume_factor = 3  # High
    elif ad_count >= 5:
        volume_factor = 2  # Medium
    else:
        volume_factor = 1  # Low
    
    # Factor in duration (longer-running campaigns suggest higher confidence)
    if ad_duration_days >= 30:
        duration_factor = 2
    elif ad_duration_days >= 7:
        duration_factor = 1.5
    else:
        duration_factor = 1
    
    combined_score = volume_factor * duration_factor
    
    if combined_score >= 7:
        return "Very High"
    elif combined_score >= 5:
        return "High"
    elif combined_score >= 3:
        return "Medium"
    else:
        return "Low"

2. Creative Strategy Analysis

We categorize competitor strategies based on observed patterns:

def analyze_creative_strategy(competitor_data):
    """
    Analyze competitor's creative strategy based on ad patterns.
    
    Args:
        competitor_data: Dictionary with competitor ad data
    
    Returns:
        Strategy classification and recommendations
    """
    ad_formats = competitor_data['ad_formats']
    ad_count = int(competitor_data['ad_count'])
    sample_ad_url = competitor_data['sample_ad_url']
    
    # Determine strategy based on format diversity
    formats_list = [f.strip() for f in ad_formats.split(',') if f.strip()]
    
    if len(formats_list) >= 3:
        strategy = "Multi-format Testing"
        recommendation = "Competitor is testing different creative formats"
    elif 'Video' in formats_list:
        strategy = "Video-First Approach"
        recommendation = "Competitor prioritizes video content"
    elif 'Carousel' in formats_list:
        strategy = "Feature Showcase"
        recommendation = "Competitor highlights multiple product features"
    else:
        strategy = "Single Creative Focus"
        recommendation = "Competitor focuses on one proven creative"
    
    # Determine testing intensity
    if ad_count >= 30:
        testing_intensity = "Aggressive Testing"
    elif ad_count >= 10:
        testing_intensity = "Active Testing"
    elif ad_count >= 3:
        testing_intensity = "Conservative Testing"
    else:
        testing_intensity = "Steady State"
    
    return {
        'strategy': strategy,
        'testing_intensity': testing_intensity,
        'recommendation': recommendation,
        'primary_format': formats_list[0] if formats_list else 'Unknown'
    }

def identify_gtm_opportunity(competitor_data, analysis):
    """
    Identify research-based outreach opportunities based on competitor weaknesses.
    Find competitor gaps to exploit.
    """
    # Look for signs of inefficiency or gaps
    if int(competitor_data['ad_count']) > 50 and analysis['testing_intensity'] == 'Steady State':
        # High volume but low innovation - opportunity for better creatives
        return {
            'opportunity': f"{competitor_data['company']} running high-volume campaigns with low innovation",
            'approach': "Position superior creative strategy as competitive advantage",
            'prospects': get_shared_prospects(competitor_data['company'])
        }

    if 'Video' not in competitor_data['ad_formats'] and competitor_data['ad_count'] > 20:
        # Missing video format while running many ads
        return {
            'opportunity': f"{competitor_data['company']} missing video format in ad mix",
            'approach': "Offer video ad expertise as differentiation",
            'prospects': get_shared_prospects(competitor_data['company'])
        }

    if competitor_data['avg_duration'] < 3:  # Less than 3 days average
        # High churn in ad creatives - possible strategy problems
        return {
            'opportunity': f"{competitor_data['company']} has high creative churn (low avg duration)",
            'approach': "Position creative stability and optimization expertise",
            'prospects': get_shared_prospects(competitor_data['company'])
        }

    return None
